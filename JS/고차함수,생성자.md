## sort() - 고차함수

### sort()문제와 한계점
* 문제점 : 일의 자리 4가 10의 자리보다 뒤쪽에 정렬
* 원인 : sort 메서드로 정렬될 때 배열의 요소가 일시적으로 문자열로 변경되어 발생

![image](https://user-images.githubusercontent.com/64582209/134908977-86bac63f-4a14-41f6-ae66-4be8f7889168.png)
* 한계점 : 대소문자 구분 없이 정렬하고 싶지만, 대소문자 구분되어 정렬

![image](https://user-images.githubusercontent.com/64582209/134909113-c9680e27-346e-407d-9f96-3df5117d63a9.png)

### 1.고차함수를 이용한 숫자 배열 정렬
```
let nums = [1, -1, 0, 4, 20, 12, 10];
console.log(nums.sort()); // [-1, 0, 1, 10, 12, 20, 4]
console.log(nums.reverse()); // [ 4, 20, 12, 10, 1, 0, -1]

let ascending_order = function(x,y){
    return x - y; // x - y > 0이면 두 수의 순서가 바뀌어 큰 수가 뒤로간다.
}

let descending_order = function(x,y){
    return y - x; // y - x > 0이면 두 수의 순서가 바뀌어 큰 수가 앞에온다.
}

console.log(nums.sort(ascending_order));
// [ -1, 0, 1, 4, 10, 12, 20]
console.log(nums.sort(descending_order));
// [ 20, 12, 10, 4, 1, 0, -1]
```
### 2. 고차함수를 이용한 문자 배열 정렬
```
let fruits = ["apple", "Orange", "orange", "melon"]
console.log(fruits.sort()); // [ 'Orange', 'apple', 'melon', 'orange' ]
console.log(fruits.reverse()); // [ 'orange', 'melon', 'apple', 'Orange' ]

let ascending_order = function(x,y){
    x = x.toUpperCase();
    y = y.toUpperCase();
   
    if(x > y) return 1; // 오름차순에서 x가y보다 크다면 x가 y보다 뒤에 와야하므로 1을 return
    else if(y > x) return -1; // 오름차순에서 y가 x보다 크다면 순서를 그대로 해야되기 때문에 -1을 return
    else return 0;
}

let descending_order = function(x,y){
    x = x.toUpperCase();
    y = y.toUpperCase();

    if(x < y) return 1; // 내림차순에서 y가x보다 크다면 y가 x보다 앞에 와야하므로 1을 return
    else if(y < x) return -1; // 내림차순에서 x가 y보다 크다면 순서를 그대로 해야되기 때문에 -1을 return
    else return 0;
}

console.log(fruits.sort(ascending_order)); // [ 'apple', 'melon', 'orange', 'Orange' ]
console.log(fruits.sort(descending_order)); // [ 'orange', 'Orange', 'melon', 'apple' ]

```
### 3. 숫자,문자 공용화
```

let ascending_order = function(x,y){
    if(typeof x === "string") x = x.toUpperCase();
    if(typeof y === "string") y = y.toUpperCase();

    return x > y ? 1 : -1
}

let descending_order = function(x,y){
    if(typeof x === "string") x = x.toUpperCase();
    if(typeof y === "string") y = y.toUpperCase();

    return x < y ? 1 : -1
}
let fruits = ["apple", "Orange", "orange", "melon"];
let nums = [1, -1, 4, 0, 12, 10, 20];
console.log(nums.sort(ascending_order)); // [ -1, 0, 1, 4, 10, 12, 20]

console.log(fruits.sort(ascending_order)); // [ 'apple', 'melon', 'orange', 'Orange' ]
console.log(fruits.sort(descending_order)); // [ 'orange', 'Orange', 'melon', 'apple' ]

```

* Array.forEach(function(item, index, array){})
* item : 배열요소, index : 배열위치, array : 배열

![image](https://user-images.githubusercontent.com/64582209/134917022-ab54cdac-0f6b-4cd3-8809-835c4e3428b5.png)

### map()
* 배열 요소 별 함수 호출 및 결과를 배열로 반환
* Array.map(function(item,index, array){})

```
// for loop
let nums = [1,2,3,4,5];
let use_for_loop = [];

for(let i = 0; i < nums.length; i++){
    use_for_loop.push(nums[i] * 2)
}
console.log(use_for_loop); // [ 2, 4, 6, 8, 10 ]

// map
let use_map = nums.map(function(item){
    return item * 2
})
console.log(use_map); // [ 2, 4, 6, 8, 10 ]
```
* map()을 이용하여 배열 요소 별로 함수를 실행시키고 각각의 결과를 새로운 배열로 반환

### find()
* 콜백 함수의 조건을 만족하는, 단 하나의 값만 반환
* Array.find(function(item,index,array){})

```
let users = [
    {name:'bob', age:17, job:false},
    {name:'elice', age:20, job:false},
    {name:'john', age:27, job:true}
]
let find_job = users.find(function(user){
    return user.job === true;
})

let find_age = users.find(function(user){
    return user.age >= 19;
})
console.log(find_job) // { name: 'john', age: 27, job: true }
console.log(find_age) // { name: 'elice', age: 20, job: false }
```
* user.age >= 19를 만족하는 객체가 john도 있지만 find는 단 하나의 값만 반환하기 때문에 먼저 조건을 만족한 elice가 반환된다.

### filter()
* 콜백 함수의 조건을 만족하는 값을 배열로 반환
* Array.filter(function(item,index,array){})

```
let users = [
    {name:'bob', age:17, job:false},
    {name:'elice', age:20, job:false},
    {name:'john', age:27, job:true}
]
let find_job = users.filter(function(user){
    return user.job === false;
})

let find_age = users.filter(function(user){
    return user.age >= 19;
})
console.log(find_job) 
/*
[
  { name: 'bob', age: 17, job: false },
  { name: 'elice', age: 20, job: false }
]
*/
console.log(find_age) 
/*
[
  { name: 'elice', age: 20, job: false },
  { name: 'john', age: 27, job: true }
]
*/
```
* find와 같은 원리이지만 filter는 조건을 만족하는 모든 값을 배열로 반환한다.

### reduce()
* 요소 별 함수 수행 누적 결과값 반환
* Array.reduce(function(accumulator, item, index, array){})
* accumulator : 이전 함수 결과(initial로 초기값 설정 가능)

```
let nums = [1,2,3,4,5]
let call_count = 0;

console.log("result\tvalue\tindex")
let sum = nums.reduce(function(accumulator, item, index, array){
    console.log(accumulator, "\t\t", item, "\t\t", index)
    call_count++;
    return accumulator + item;
})
/*
result	value	index
1 		 2 		 1
3 		 3 		 2
6 		 4 		 3
10 		 5 		 4
*/

console.log(call_count); // 4, 배열의 요소는 5개지만 초기값을 지정하지 않으면 최초 accumulator 값에는 0번째 index값이 들어가기 때문에 index1부터4까지만 4번만 수행한다.
console.log(sum) // 15

```

# 생성자 함수
* 유사한 객체를 다중으로 만들 때 사용되는 함수(class 개념과 유사)
* 일반적으로 생성자 함수의 첫 글자는 대문자
* 생성자 함수로 객체 생성시 new 연산자를 통해 객체 생성

## new.target
* 유연한 생성자 함수를 만들기 위해 new.target이 없을때 new키워드를 추가하여 호출되도록 처리

```
function FishBread(flavor, price){
    if(!new.target){
        return new FishBread(flavor, price);
    }
    this.flavor = flavor;
    this.price = price;
}

const bread_1 = new FishBread("milk", 1200);
const bread_2 = new FishBread("cream", 1300);
const bread_3 = FishBread("redbean", 1500)
console.log(bread_1); // FishBread { flavor: 'milk', price: 1200 }
console.log(bread_2); // FishBread { flavor: 'bean', price: 1300 }
console.log(bread_3); // FishBread { flavor: 'redbean', price: 1500 }
```
* 생성자 함수 FishBread로 틀을 만들고 new연산자를 통해 손쉽게 새로운 객체 생성
* new를 붙이지 않고 bread_3을 선언해도 new.target처리를 통해 new가 없어도 객체가 생성됨
