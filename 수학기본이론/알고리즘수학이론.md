# 알고리즘에 사용되는 수학 이론

## 알고리즘 평가 지표
* 코딩테스트에서는 메모리 사용량과 시간 복잡도가 제일 중요하다.
![image](https://user-images.githubusercontent.com/64582209/135389957-f45ff3a0-9657-4ee0-8e53-6e4237cffd89.png)

## 시간 복잡도
* 입력 크기의 값에 대해 단위 연산을 몇 번 수행하는지 계산하여, 알고리즘의 수행시간을 평가
* 3가지 점근적 표현법
  * O(빅오): 최악의 상황을 고려하여 성능 측정 결과 표현
  * Θ(세타): 평균적인 경우에서의 성능 측정 결과 표현
  * Ω(오메가): 최선의 상황일 때의 성능 측정 결과 표현

## 경우의 수
* 어떤 사건 혹은 일이 일어날 수 있는 경우의 가짓수를 수로 표현
* 완전탐색으로 경우의 수를 푸는 알고리즘
  * 순열 : 서로 다른 n개의 원소중 r을 중복 없이 골라 순서에 상관 있게 나열하는 경우의 수(nPr)
  * 조합 : 서로 다른 n개의 원소중 r을 중복 없이 골라 순서에 상관 없이 나열하는 경우의 수(nCr)
  * 중복 순열 : 서로 다른 n개의 원소중 r을 중복 있게 골라 순서에 상관 있게 나열하는 경우의 수(nH)

### 순열 예제
1. for loop을 이용한 풀이
* 배열의 요소가 많아질수록 for문과 if문이 많아지기 때문에 많이 복잡해지고 느려짐

```
let input = ["a","b","c"]
let count = 0

function permutation(arr){
    // i는 첫번째 index에 위치시킬 요소
    for(let i = 0; i < arr.length; i++){
        // j는 두번째 index에 위치시킬 요소
        for(let j = 0; j < arr.length; j++){
            if(i == j) continue // i에 위치한 요소와 j의 위치한 요소가 중복일 경우 패스
            // k는 세번째 index에 위치시킬 요소
            for(let k = 0; k < arr.length; k++){
                if(k == i) continue;
                if(k == j) continue;

                console.log(arr[i], arr[j], arr[k])
                count++
            }
        }
    }
}

permutation(input)
console.log(count)
/*
[ 'a', 'b', 'c' ]
[ 'a', 'c', 'b' ]
[ 'b', 'a', 'c' ]
[ 'b', 'c', 'a' ]
[ 'c', 'b', 'a' ]
[ 'c', 'a', 'b' ]
6
*/
```
2. 재귀함수
```
let input = ["a","b","c"]
let count = 0

function permutation(arr, s, r){
    
    if(s == r){
        count++
        console.log(arr)
        return
    }
    
    for(let i = s; i < arr.length; i++){
        [arr[s],arr[i]] = [arr[i],arr[s]];
        permutation(arr, s + 1, r);
        [arr[s],arr[i]] = [arr[i],arr[s]];
    }
}

permutation(input, 0 , 2)
console.log(count)
/*
[ 'a', 'b', 'c' ]
[ 'a', 'c', 'b' ]
[ 'b', 'a', 'c' ]
[ 'b', 'c', 'a' ]
[ 'c', 'b', 'a' ]
[ 'c', 'a', 'b' ]
6
*/
```

### 조합 예제
1. for loop

```
let input = [1,2,3,4]
let count = 0

function combination(arr){
    for(let i = 0; i < arr.length; i++){
        for(let j = i + 1; j < arr.length; j++){
            count++
            console.log(arr[i],arr[j])
        }
    }
}

combination(input)
console.log(count)
/*
1 2
1 3
1 4
2 3
2 4
3 4
6
*/
```

2. 재귀함수
```
let input = [1,2,3,4];
let output = [];
let count = 0;

function combination(arr, data, s, idx, r){
    if(s == r){
        count++;
        console.log(data);
        return;
    }
    for(let i = idx; arr.length - i >= r - s; i++){
        data[s] = arr[i];
        combination(arr, data, s + 1, i + 1, r);
    }
}

combination(input, output, 0, 0, 2);
console.log(count);
/*
1 2
1 3
1 4
2 3
2 4
3 4
6
*/
```
