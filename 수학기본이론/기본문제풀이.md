## 1. 등차수열의 항 찾기
![image](https://user-images.githubusercontent.com/64582209/135421688-f44d0c7e-64a2-48cb-a53f-a3e9de961bb1.png)

### 풀이
```
function answer(a,d,n){
    let index = -1;
    /* 1번풀이
    let result;
    // 무한루프 반목문 만들어서 조건을 걸어 break하기
    for(let i = 1; ; i++){
        result = a + d * (i-1) // 등차수열 식 만들기
        if(result == n) {
            index = i
            break
        }
        if(result > n){
            break
        }
    }
    */
    // 2번 풀이
    if((n - a) % d == 0){
        index = (n - a) / d +1
    }
    
    

    return index
}

let input = [
    [1,2,7],
    [2,3,10],
    [3,5,23]
]

for(let i = 0; i < input.length; i++){
    console.log(`#${i + 1} ${answer(input[i][0], input[i][1], input[i][2])}`)
}
```

## 2. 잃어버린 카드 찾기
![image](https://user-images.githubusercontent.com/64582209/135435452-e7e7b20a-df0e-49f8-a762-dbccbd8c8857.png)
### 풀이
```
function answer(a,b,c){
    let number = 0;

    num = [a,b,c];
    num.sort((x,y) => x - y);
    // 등차수열의 등차값 찾기
    let d = 0;
    for(let i = 1; i < num.length; i++){
        d += num[i] - num[i - 1]
    }
    d /= num.length
    //  빈 idx 찾기
    let index = num[2] - num[1] > num[1] - num[0] ? 2 : 1

    // 비어있는 숫자 찾기
    number = num[0] + d * index

    return number

}

let input = [
    [1,7,10],
    [3,8,18],
    [11,5,2]
]

for(let i = 0; i < input.length; i++){
    console.log(`#${i + 1} ${answer(input[i][0], input[i][1], input[i][2])}`)
}
```
