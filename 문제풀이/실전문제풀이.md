## 2. 짝수와 홀수
![image](https://user-images.githubusercontent.com/64582209/135446799-4a6360fb-a5e4-48bf-876f-827b1b1c8968.png)
### 나의 풀이
```
function solution(num) {
    var answer = '';
    if(num % 2 == 0){
        answer = 'Even'
    }else if(num % 2 !== 0){
        answer = 'Odd'
    }
    return answer;
}

console.log(solution(3)) // Odd
console.log(solution(4)) // Even
```
### 다른 풀이
```
function solution(num){
    return num % 2 ? "Odd" : "Even"
}

console.log(solution(3)) //Odd
console.log(solution(4)) // Even
```

## 4. 폰켄몬
![image](https://user-images.githubusercontent.com/64582209/135447400-5e24421f-a1fd-4c66-b42a-e619ce53b921.png)

### 나의 풀이
```
function solution(nums){
    // 뽑을수 있는 폰켓몬 마리수
    let num = nums.length / 2
    // 폰켄몬 종류의 수
    let kind = new Set(nums).size


    return kind > num ? num : kind
}

console.log(solution([3,2,3,1,2,4])) // 3
```

## 5. 가운데 글자 가져오기
![image](https://user-images.githubusercontent.com/64582209/135450863-fd8144ea-ace7-4a5f-8f32-dc6deef874cf.png)

### 나의 풀이
```
function solution(str){
    const lengthDiv = str.length / 2
    // 홀수
    if(str.length % 2 != 0){
        return str[Math.floor(lengthDiv)]
    }else{
    // 짝수
        return str.slice(lengthDiv - 1, lengthDiv + 1)
    }

}

console.log(solution("abcdef")) // cd
```

## 6. 문자열 내 마음대로 정렬하기
![image](https://user-images.githubusercontent.com/64582209/135453749-13fcfd6d-2b93-49b2-a5f5-613d107dd668.png)
### 나의 풀이
```
function solution(arr, n){
    // 배열 요소들의 n번째 index 값
        arr.sort(function(a,b){
            if(a[n] > b[n]){
                return 1
            }else if(a[n] < b[n]){
                return -1
            }else if(a[n] == b[n]){
                if(a > b){
                    return 1
                }else{
                    return -1
                }
            }
        })
    return arr
}

console.log(solution(["abce", "abcd", "cdx"],1)) // [ 'abcd', 'abce', 'cdx' ]

```
### 다른 풀이
```
1.
function solution(arr,n){
    return arr.sort((a,b) => a[n] == b[n] ? (a > b ? 1 : -1) : (a[n] > b[n] ? 1 : -1))
}

console.log(solution(["abce", "abcd", "cdx"],1)) //[ 'abcd', 'abce', 'cdx' ]

function solution(strings, n) {
    // strings 배열
    // n 번째 문자열 비교
    return strings.sort((s1, s2) => s1[n] === s2[n] ? s1.localeCompare(s2) : s1[n].localeCompare(s2[n]));
}
```

## 8. 두 정수 사이의 합
![image](https://user-images.githubusercontent.com/64582209/135463567-5f8f06af-7184-42d1-b91b-763fd95e2514.png)

### 나의 풀이
```
function solution(a,b){
    let sum = 0
    for(let i = Math.min(a,b); i <= Math.max(a,b); i++){
        sum += i
    }
    return sum
}

console.log(solution(3,5)) // 12
```

### 다른 풀이

```
function solution(a,b){
```

## 9. 부족한 금액 계산하기

```
function solution(price, money, count){

    let sum = 0
    for(let i = 1; i <= count; i++){
        sum += price*i
    }

    
    return sum > money ? sum - money : 0
}
```

## 10. K번째 수
![image](https://user-images.githubusercontent.com/64582209/135475346-4d3c55c5-1e12-4441-b1d2-7a83eff9b296.png)

### 나의 풀이
```
function solution(array, commands){
    let result = []
    for(let i = 0; i < commands.length; i++){
        const start = commands[i][0] - 1
        const end = commands[i][1]
        const arr_index = commands[i][2] - 1
        let arr = array.slice(start, end)
        arr.sort((x,y) => x - y)
        result.push(arr[arr_index])
    }
    return result
}

console.log(solution([1,5,2,6,3,7,4],[[2,5,3],[4,4,1],[1,7,3]]))
```

```
function solution(arr, divisor) {
    const array = arr.filter((item) => {
        return item % divisor == 0
    })

    return array.length === 0 ? [-1] : array.sort((x,y) => x - y)
    
}
```

### 12. 같은 숫자는 싫어
![image](https://user-images.githubusercontent.com/64582209/135479887-947a10dc-d899-4c4f-b5ca-8432d405b365.png)

### 나의 풀이
```
function solution(arr){
    let result = []
    for(let i = 0; i < arr.length ; i++){
        if(arr[i] == arr[i + 1]) continue
        else result.push(arr[i])
    }
    return result
}

console.log(solution([1,1,3,3,0,1,1]))
```

### 다른 풀이
```
function solution(arr){
    return arr.filter((val, index) => val != arr[index + 1])
}
console.log(solution([1,1,3,3,0,1,1]))
```
