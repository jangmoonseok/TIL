## 2. 짝수와 홀수
![image](https://user-images.githubusercontent.com/64582209/135446799-4a6360fb-a5e4-48bf-876f-827b1b1c8968.png)
### 나의 풀이
```
function solution(num) {
    var answer = '';
    if(num % 2 == 0){
        answer = 'Even'
    }else if(num % 2 !== 0){
        answer = 'Odd'
    }
    return answer;
}

console.log(solution(3)) // Odd
console.log(solution(4)) // Even
```
### 다른 풀이
```
function solution(num){
    return num % 2 ? "Odd" : "Even"
}

console.log(solution(3)) //Odd
console.log(solution(4)) // Even
```

```
function solution(a, b) {
```

## 4. 폰켄몬
![image](https://user-images.githubusercontent.com/64582209/135447400-5e24421f-a1fd-4c66-b42a-e619ce53b921.png)

### 나의 풀이
```
function solution(nums){
    // 뽑을수 있는 폰켓몬 마리수
    let num = nums.length / 2
    // 폰켄몬 종류의 수
    let kind = new Set(nums).size


    return kind > num ? num : kind
}

console.log(solution([3,2,3,1,2,4])) // 3
```

## 5. 가운데 글자 가져오기
![image](https://user-images.githubusercontent.com/64582209/135450863-fd8144ea-ace7-4a5f-8f32-dc6deef874cf.png)

### 나의 풀이
```
function solution(str){
    const lengthDiv = str.length / 2
    // 홀수
    if(str.length % 2 != 0){
        return str[Math.floor(lengthDiv)]
    }else{
    // 짝수
        return str.slice(lengthDiv - 1, lengthDiv + 1)
    }

}

console.log(solution("abcdef")) // cd
```

### 다른 풀이
