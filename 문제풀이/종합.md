## 1. 최소값 구하기
![image](https://user-images.githubusercontent.com/64582209/135272204-7baf6a37-d6c6-474f-973d-d9dddc2d9abb.png)
### 풀이
```
function answer(x,y){
    let min
    /* 내 풀이
    min = Math.min(x,y)
    */
    min = x > y ? y : x
    return min
}

let input = [
    [16,3],
    [-3,1],
    [1000,525]
]

for(let i = 0; i < input.length; i++){
    console.log(`#${i + 1} ${answer(input[i][0], input[i][1])}`)
}
```

## 2. 제곱 구현
![image](https://user-images.githubusercontent.com/64582209/135273139-60e44c80-8cc6-489d-b21e-eebc288b2297.png)

### 풀이
```
function answer(x,y){
    let result = 1
    /* 내풀이
    result = Math.pow(x,y)
    */
    for(let i = 0; i < y; i++){
        result *= x
    }
    return result
}

let input = [

    [4,6],
    [1,100]
]
```

## 4. 요일 구하기
![image](https://user-images.githubusercontent.com/64582209/135274370-215e88fd-ed93-4359-a4d5-4b4f31e19f0a.png)

### 풀이
```
function answer(str){
    let week = new Array(
        "일요일",
        "월요일",
        "화요일",
        "수요일",
        "목요일",
        "금요일",
        "토요일"
    )
    let day
    /* 내풀이
    let date = new Date(str).getDay()
    for(let i = 0; i < week.length; i++){
        if(i === date) day = week[i]
    }
    */

    let date = new Date(str).getDay()
    day = week[date]
    
    
    return day
}

let input = [

    "2021,02,27",
    "2021-03-14"
]


for(let i = 0; i < input.length; i++){
```

## 6. 배열 내 최대값 구하기
![image](https://user-images.githubusercontent.com/64582209/135279818-6bb4afdc-2d12-4e75-8187-ca41c061a617.png)

### 풀이
```
function answer(arr){
    let max
    for(let i = 0; i < arr.length; i++){
        if(!max){
            max = arr[i]
        }
        else if(max < arr[i]){
            max = arr[i]
        }
    }

    /* Math.max 이용
    max = Math.max.apply(null, arr)
    */
    return max
}

let input = [
    [1,6,5,2,3],
    [19,41,23,-4,17],
    [-64,-27,-41,-33,-59]
]


for(let i = 0; i < input.length; i++){
```

## 9. 문자 교정
![image](https://user-images.githubusercontent.com/64582209/135285512-b3aa1318-cfa8-4149-8cbe-b1500f1d3124.png)
```
function answer(str){
    let fix_str = ""
    /* 
    1. 단어 기준으로 나눈다. => 공백 기준으로 나눈다.
    2. 단어의 앞 철자를 대문자로 변환시켜준다,.
    3. 다시 문자열르 변환한다.
    */
    let str_split = str.split(" ")
    for(let item of str_split){
        fix_str += item[0].toUpperCase() + item.slice(1) + " "
    }
    return fix_str
}

let input = [
    "Hello, My Name is john",
    "This week is closed due to COVID-19",
    "fifty percent off this week"
]


for(let i = 0; i < input.length; i++){
    console.log(`#${i + 1} ${answer(input[i])}`)
}
```

## 10. 2차원 배열의 곱셈 
![image](https://user-images.githubusercontent.com/64582209/135288488-32521d78-4296-4351-b642-47f8f76346a0.png)
### 풀이
```
function answer(arr){
    let product = 1
    for(let i = 0; i < arr.length; i++){
        for(let j = 0; j < arr[i].length; j++){
            product *= arr[i][j]
        }
    }
    return product
}

let input = [
    [[1],[2],[3]],
    [
        [1,2],
        [3,4],
        [5,6,7]
    ],
    [
        [5,1],
        [0.2, 4, 0.5],
        [3,9]
    ]
]


for(let i = 0; i < input.length; i++){
    console.log(`#${i + 1} ${answer(input[i])}`)
}
```
