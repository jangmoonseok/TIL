# 반복문 문제 풀이

## 1. 별별별
![image](https://user-images.githubusercontent.com/64582209/135248978-39dd7760-335e-4a10-96d3-4ec72a547326.png)

### 풀이
```
function answer(num){
    let result = ""

    for(let i = 0; i < num; i++){
        result += "*"
    }
    return result
}

let input = [5,7,12]

for(let i = 0; i < input.length; i++){
    console.log(`#${i + 1} ${answer(input[i])}`)
}
/*
output:
#1 *****
#2 *******
#3 ************
*/
```
## 2. 두 수 사이 숫자
![image](https://user-images.githubusercontent.com/64582209/135249423-404671d4-6772-4dd5-813c-d9e053ae6581.png)
### 풀이
```
function answer(x,y){
    let result = []
    /*
    나의 풀이
    if(x > y){
        for(let i = 0; i < x - y + 1; i++){
            result.push(y + i)
        }
    }else{
        for(let i = 0; i < y - x + 1; i++){
            result.push(x + i)
        }
    }
    */
    //  x가 y보다 큰 수가 들어온 경우 자리를 바꿔주는 조건문
    if(x > y) {
        let t = x
        x = y
        y = t
    }
    for(let i = x; i <= y; i++){
        result.push(i)
    }

    

    return result
}

let input = [[3,7],[8,3],[12,10]]

for(let i = 0; i < input.length; i++){
    console.log(`#${i + 1} ${answer(input[i][0], input[i][1])}`)
}
/*
#1 3,4,5,6,7
#2 3,4,5,6,7,8
#3 10,11,12
*/
```

## 3. 반 평균
![image](https://user-images.githubusercontent.com/64582209/135252660-e27038eb-8af2-44c0-94d7-a9f57e8704f3.png)

### 풀이
```
function answer(score){
    let average = 0;

    let sum = 0;
    for(let i = 0; i < score.length; i++){
        sum += score[i]
    }
    average = (sum / score.length).toFixed(2)
    return average;
}

let input = [
    [80,95,65,70,100],
    [82,77,51,64,73,90,80],
    [100,71,59,88,72,75,91,93]
]

for(let i = 0; i < input.length; i++){
    console.log(`#${i + 1} ${answer(input[i])}`)
}
```
## 4. 핸드폰 판매
![image](https://user-images.githubusercontent.com/64582209/135254342-016c723e-9efc-4618-8ef9-79634fc30cdd.png)
```
function answer(employee){
    let employee_id
    let maxNum = 0
    for(let i = 0; i < employee.length; i++){
        if(maxNum < employee[i]){
            maxNum = employee[i]
            employee_id = i + 1
        }
    }
    return employee_id
}

let input = [
    [3,7,9,6,1],
    [2,7,1,4,3,0,5],
    [7,5,0,1,2,12,6]
]

for(let i = 0; i < input.length; i++){
    console.log(`#${i + 1} ${answer(input[i])}`)
}
```

## 5. 무한 뺄셈
![image](https://user-images.githubusercontent.com/64582209/135259155-e3ce5369-3a10-46a8-8d6d-789bcaf4821c.png)
### 풀이
```
function answer(s,e){
    let sequence = []
    sequence.push(s)
    sequence.push(e)
    
    let sum;
    while(1){
        sum = s - e
        s = e
        e = sum
        if(sum < 0 ) break
        sequence.push(sum)
    }
    return sequence
}

let input = [
    [9,3],
    [6,3],
    [13,7]
]

for(let i = 0; i < input.length; i++){
    console.log(`#${i + 1} ${answer(input[i][0], input[i][1])}`)
}

```
