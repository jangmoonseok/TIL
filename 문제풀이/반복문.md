# 반복문 문제 풀이

## 1. 별별별
![image](https://user-images.githubusercontent.com/64582209/135248978-39dd7760-335e-4a10-96d3-4ec72a547326.png)

### 풀이
```
function answer(num){
    let result = ""

    for(let i = 0; i < num; i++){
        result += "*"
    }
    return result
}

let input = [5,7,12]

for(let i = 0; i < input.length; i++){
    console.log(`#${i + 1} ${answer(input[i])}`)
}
/*
output:
#1 *****
#2 *******
#3 ************
*/
```
## 2. 두 수 사이 숫자
![image](https://user-images.githubusercontent.com/64582209/135249423-404671d4-6772-4dd5-813c-d9e053ae6581.png)
### 풀이
```
function answer(x,y){
    let result = []
    /*
    오답
    if(x > y){
        for(let i = 0; i < x - y + 1; i++){
            result.push(y + i)
        }
    }else{
        for(let i = 0; i < y - x + 1; i++){
            result.push(x + i)
        }
    }
    */
    //  x가 y보다 큰 수가 들어온 경우 자리를 바꿔주는 조건문
    if(x > y) {
        let t = x
        x = y
        y = t
    }
    for(let i = x; i <= y; i++){
        result.push(i)
    }

    

    return result
}

let input = [[3,7],[8,3],[12,10]]

for(let i = 0; i < input.length; i++){
    console.log(`#${i + 1} ${answer(input[i][0], input[i][1])}`)
}
/*
#1 3,4,5,6,7
#2 3,4,5,6,7,8
#3 10,11,12
*/
```
