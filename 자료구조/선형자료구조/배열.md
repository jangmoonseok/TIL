## some()
* 배열 내 단 하나라도 콜백 함수의 조건을 만족하는 요소가 있다면 true, 아니면 false반환( 빈 배열이면 false)
* Array.some(function(item,index,array){})
```
let users = [
    {name:"bob", age:17, job:false},
    {name:"alice", age:20, job:false},
    {name:"john", age:27, job:true}
]

let some_job = users.some(user => {
    return user.job == false
})
console.log(some_job) // true

let some_age = users.some(user => {
    return user.age < 16
})
console.log(some_age) // false

let empty = [].some(item => item > 16)
console.log(empty) // false
```


## every()
* 배열 내 모든 요소가 콜백 함수의 조건을 만족한다면 true, 아니면 false( 빈 배열인 경우 true)
* Array.every(function(item,index,arr){})

```
let users = [
    {name:"bob", age:17, job:false},
    {name:"alice", age:20, job:false},
    {name:"john", age:27, job:true}
]

let some_job = users.every(user => {
    return user.job == false
})
console.log(some_job) // true

let some_age = users.every(user => {
    return user.age > 16
})
console.log(some_age) // false

let empty = [].every(item => item > 16)
console.log(empty) // false
```

function answer(nums){
    let result = [];
    // 최소값 구하기 //
    // let min_num = Math.min.apply(null, nums) //
    let min_num = Number.MAX_SAFE_INTEGER;
    for(let i = 0; i< nums.length; i++){
        if(min_num > nums[i]){
            min_num = nums[i]
        }
    }
    // min을가진 index를 배열로 반환 //
    for(let i = 0; i < nums.length; i++){
        if(nums[i] == min_num) result.push(i)
    }
    console.log(result)
}

let input = [
    [5,2,10,2],
    [4,5,7,4,8],
    [12,11,11,16,11,12]
]

for(let i = 0; i < input.length; i++){
    answer(input[i])
}
/*
[ 1, 3 ]
[ 0, 3 ]
[ 1, 2, 4 ]
*/

        nums.splice(nums.indexOf(max),1)

  */
    let result = []

    result = nums[0] > nums[1] ? [nums[0], nums[1]] : [nums[1], nums[0]]

            result[0] = nums[i]

            result[1] = nums[i]
        }
    }
    console.log(result)
}

#### 풀이
```
function answer(nums, target){
    let result = []
    /* O(n^2)
    for(let i = 0; i < nums.length; i++){
        for(let j = i + 1; j < nums.length; j++){
            if(nums[i] + nums[j] == target){
                result.push(i,j)
            }   
        }
    }
    */
    // O(n)
    for(let i = 0; i < nums.length; i++){
        let count = 1
        if(nums[i] + nums[count] != target){
            count++
        }else{
            result.push(i, count)
            break
        }

    }
    console.log(result)
}

let input = [
    [[2,7,11,15],9],
    [[3,2,4],6],
    [[3,3],6]
]

for(let i = 0; i < input.length; i++){
    answer(input[i][0],input[i][1])
}
```

```
function answer(blocks){
    let result = 0

    // 1. 블록의 전체 개수를 칸만큼 나누기
    let avg = 0
    for(let i = 0; i < blocks.length; i++){
        avg += blocks[i]
    }
    avg /= blocks.length
    // 2. 그 칸의 벽돌 수 < sum 이라면 부족한 벽돌수 계산
    for(let i = 0; i < blocks.length; i++){
        if(blocks[i] < avg){
            result += avg - blocks[i]
        }
    }

    console.log(result)
}

let input = [
    [5,2,4,1,7,5],
    [12,8,10,11,9,5,8],
    [27,14,19,11,26,25,23,15]
]

    answer(input[i])

function answer(s, e){
    let result = []
    // result의 모든 인덱스를 0으로 초기화
    for(let i = 0; i < 10; i++){
        result[i] = 0
    }
    // s부터 e까지 반복문
    let num
    for(let i = s; i <= e; i++){
        num = i
        while(num != 0){
            result[num % 10]++
            num /= 10
            num = parseInt(num)
        }
    }

    console.log(result)
}

#### 풀이
```
function answer(length){
    let result = []
    // 1.result를 2중배열로 만들기
    for(let i = 0; i < length; i++){
        result[i] = []
    }
    // 2. 패턴대로 반복문 구현
    let direction = 1
    let x,y,num
    y = num = 0
    x = -1
    while(1){
        // 첫줄은 length길이만큼 시작해서 숫자를 채운다 
        for(let i = 0; i < length; i++){
            x += direction
            result[y][x] = ++num
        }
        // length - 1
        length--

        if(length == 0) break
        // length - 1이 된 상태로 한번 방향이 바뀌고 한번
        for(let i = 0; i < length; i++){
            y += direction
            result[y][x] = ++num
        }

        direction *= -1
    }
    console.log(result)
}

let input = [
    3,5,6
]

for(let i = 0; i < input.length; i++){
    answer(input[i])
}
