package ch4;

import java.util.Scanner;

public class Ex4_2 {
	
	public static void main(String[] args) {		
		/*
		 * 조건식(if - else if문)
		 * - 여러 개의 조건식을 포함한 조건식
		 * if(조건식){
		 *	조건식1이 참일 때 수행
		 * }else if(조건식2){
		 * 	조건식1이 거짓이고 조건식2가 참일 때 수행
		 * }else if(조건식3){
		 * 	조건식1,2가 거짓이고 조건식3이 참일 때 수행
		 * }else{
		 * 조건식1,2,3 모두 거짓일 때 수행
		 * }
		 * - else블럭은 생략가능
		 */
		
		int score = 0;
		char grade;
		
		System.out.println("점수를 입력하세요>");
		Scanner sc = new Scanner(System.in);
		score = sc.nextInt();
		
		//방법1
		if(score >= 90) {
			grade = 'A';
		}else if(score >= 80) {
			grade = 'B';
		}else if(score >= 70) {
			grade = 'C';
		}else {
			grade = 'D';
		}
		
		//방법2 : 리팩토링
		grade = 'D';
		if(score >= 90) {
			grade = 'A';
		}else if(score >= 80) {
			grade = 'B';
		}else if(score >= 70) {
			grade = 'C';
		}
		
		System.out.println("당신의 학점은 " + grade + "입니다.");
		//결과1 : score = 80 >> 당신의 학점은 B입니다.
		//결과2 : score = 73 >> 당신의 학점은 C입니다.
		
		//중첩 if문 : if문 안의 if문
		int num = 0;
		char sign;
		//(1) : 괄호 생략한 중첩if문
		if(num >= 0)
			if(num != 0)
				sign = '+';
		else
			sign = '-';
		//(2)
		if(num >= 0) {
			if(num != 0) {
				sign = '+';
			}
			else {
				sign = '-';
			}
		}
		
		//(1)문장에서 들여쓰기로 보면 (2)문장과 달라보이지만 else문이 한개만 있다면 가장 가까운if문과 함께 쓰인다.
	}
}
