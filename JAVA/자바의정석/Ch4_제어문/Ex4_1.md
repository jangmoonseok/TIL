package ch4;

public class Ex4_1 {
	/*
	 * 조건문(if문)
	 * - 조건식이 참(true)일때, 블럭{}안의 문장들을 수행한다.
	 * if(조건식) {
	 *  //조건식이 참(ture)일때 수행될 문장들을 적는다
	 * }
	 * - 조건식에 만족하지 않으면 무시하고 다음 문장부터 실행된다.
	 * 
	 * 
	 * 조건문(if - else문)
	 * 조건식이 참일 때와 거짓일 때 나눠서 처리
	 * if(조건식){
	 * 	//참일 때
	 * }else{
	 * 	//거짓일 때
	 * }
	 */
	public static void main(String[] args) {		
		int score = 70;
		if(score > 60) {
			System.out.println("합격입니다"); 
		} //조건에 만족하므로 블럭안의 문장 출력
		
		score = 60;
		if(score > 60) {
			System.out.println("합격입니다.");
		}else {
			System.out.println("탈락입니다.");
		} // 조건이 false이므로 else문에 있는 문장이 실행된다.
	}
}
