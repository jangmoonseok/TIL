package ch7;

public class ch7_24_참조변수의형변환1 {

	public static void main(String[] args) {
		/*
		 * 참조변수의 형변환
		 * - 사용할 수 있는 멤버의 갯수를 조절하는 것
		 * - 조상 자손 관계의 참조변수는 서로 형변환 가능
		 * - 참조변수(리모컨)을 변경함으로써 사용할 수 있는 멤버의 개수를 조절하기 위해서
		 */
		FireEngine f = new FireEngine();
		
		Car c = (Car)f; // OK, 조상인 Car타입으로 형변환(생략가능)
		FireEngine f2 = (FireEngine)c; // OK, 자손인 FireEngine타입으로 형변환으로 불안전한 형변환(생략불가)
//		Ambulance a = (Ambculance)f; // 에러, 상속관계가 아닌 형제 클래스 간의 형변환 불가
//		c.water(); // 에러, FireEngine타입을 조상인 Car타입으로 형변환 했으므로 FireEngine인스턴스의 water()메서드는 사용불가
		f2.water(); // OK
		
		Car c2 = new Car();
		FireEngine f3 = (FireEngine)c2; //Car인스턴스의 멤버개수 4개 -> FireEngine타입 인스턴스 멤버개수 5개 이므로 실행시 형변환 실행 에러 발생
		f3.water();
		
	}
	/*
	 * instanceof 연산자
	 * - 참조변수의 형변환 가능여부 확인에 사용, 가능하면 true반환
	 * - 형변환 전에 반드시 instanceof로 확인해야 함
	 * - 형변환할 참조변수의 클래스의  조상 클래스는 모두 형변환 가능
	 */
	void doWork(Car c) { // 참조변수 c에는 Car클래스의 자손들이 모두 들어올 수 있다
		if(c instanceof FireEngine) { // c로 Car의 자손 클래스가 들어온다면 형제끼리는 형변환을 할 수 없기때문에 instanceof로 가능여부 확인
			FireEngine fe = (FireEngine)c;
			fe.water();
		}
	}
}
class Car{

	void water() {
		System.out.println("water");
	}
